function mark_all_cell!(r::Robot)
    num_vertical = moves!(r, Sud)
    num_horizontal = moves!(r, West)
    #УТВ: Робот - в Юго-Западном углу

    num_len = moves!(r, Ost)
    #УТВ: сколько клеток составляет длина поля

    for i in 1:(num_len+1) 
        putmarkers!(r, Nord)
        putmarker!(r)
        move!(r, West)
        putmarkers!(r, Sud)
        putmarker!(r)
        if  isborder(r,West)==false   
            move!(r, West)
        end
        i=i+2
    end
    #УТВ: все клетки поля промакированы

    while isborder(r,Sud)==false    
        move!(r, Sud)
    end
    while isborder(r,West)==false
        move!(r, West)
    end
    #УТВ: Робот - в Юго-Западном углу

    moves!(r, Nord, num_vertical)
    moves!(r, Ost, num_horizontal)
    #УТВ: Робот - в исходном положении
end

function moves!(r::Robot, side::HorizonSide)
    num_steps=0
    while isborder(r,side)==false
        move!(r,side)
        num_steps+=1
    end
    return num_steps
end

function moves!(r::Robot,side::HorizonSide,num_steps::Int)
    for _ in 1:num_steps       # символ "_" заменяет фактически не используемую переменную
        move!(r,side)
    end
end

function putmarkers!(r::Robot, side::HorizonSide)
    while isborder(r,side)==false
        move!(r,side)
        putmarker!(r)
    end
end
